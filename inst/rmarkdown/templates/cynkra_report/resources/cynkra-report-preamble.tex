  \usepackage{eurosym}

  \makeatletter

     \renewcommand\allcapsspacing[1]{{\addfontfeature{LetterSpace=15}#1}}
     \renewcommand\smallcapsspacing[1]{{\addfontfeature{LetterSpace=10}#1}}

  % euro symbol, handled by xelatex

  % fruttiger and dejavue sans mono
    \setmainfont{FrutigerNeueLT}[

% for protrusion (even margins) and expansion (fewer/better hyphenations)
\hyphenpenalty=7500
% https://tex.stackexchange.com/a/335088/8057
\pretolerance=5000
\tolerance=9000
% Do not set \lefthyphenmin, \righthyphenmin, or \<lang>hyphenmins
% (https://tex.stackexchange.com/a/235279/8057):
% These values add too much space between words.

\setlength{\columnsep}{10mm} % Column separation width

\tabulinesep=1.1mm


% omit default title page
\renewcommand\maketitle{}

% Required for indentation fix and polyglossia caption setup
\usepackage{etoolbox}

% avoid small caps in french figure and table captions
\ifnum 0\ifxetex 1\fi\ifluatex 1\fi=0 % if pdftex
  \babelprovide{french}
  \renewcommand\frenchfigurename{Figure}
  \renewcommand\frenchtablename{Table}
\else
  \gappto\captionsfrench{
    \renewcommand{\figurename}{Figure}
    \renewcommand{\tablename}{Table}
  }
\fi

% Customize size, spacing, and color of headings

\titlespacing*{\chapter}{0pt}{2\baselineskip}{2\baselineskip}

\titlespacing*{\section}{0pt}{1.25\baselineskip}{0.65\baselineskip}
\titleformat{\subsection}{\normalfont\bfseries\color{cynkrablue}\fontsize{13}{15}\selectfont}{\thesubsection}{1em}{}
\titleformat{\subsubsection}{\normalfont\color{cynkrablue}\fontsize{12}{14}\selectfont}{\thesubsection}{1em}{}

% Make bullet points smaller and blue
